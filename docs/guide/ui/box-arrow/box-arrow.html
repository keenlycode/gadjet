Usage

js <script defer>

import { BoxArrow } from 'gadjet';

BoxArrow.define('el-box-arrow');
BoxArrow.tagStyle({
    arrow: 'top', // arrow direction: top, right, bottom or left
    arrowSize: '8px',
    position: '50%', // arrow position on it's side.
    bgColor: '#ccc',
    borderWidth: '1px',
    borderStyle: 'solid',
    borderColor: 'black',
    borderRadius: '4px'
})

BoxArrow.classStyle('right', {arrow: 'right'});
